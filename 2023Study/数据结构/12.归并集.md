```java
public class UnionFind {
    public static class Node<V> {
        V value;
        public Node(V value) {
            this.value = value;
        }
    }
    public static class UnionSet<V> {
        HashMap<V, Node<V>> nodes; // 包裹，用户节点。用户不察觉到
        HashMap<Node<V>, Node<V>> parents; // 代表指针，Key->Value，value为父节点
        HashMap<Node<V>, Integer> sizeMap; // 存放“代表节点”的包含节点的数量。
        public UnionSet(List<V> values) {
            nodes = new HashMap<>();
            parents = new HashMap<>();
            sizeMap = new HashMap<>(); //根据数量可以知道有几个集合。只留代表节点记录。
            for (V value : values) {
                Node cur = new Node(value);
                nodes.put(value, cur);
                parents.put(cur, cur);
                sizeMap.put(cur, 1);
            }
        }
        boolean isSameSet(V a, V b) {
            return findFather(a) == findFather(b);
        }
        void union(V a, V b) {
            Node na = findFather(a);
            Node nb = findFather(b);
            if (isSameSet(a, b)) {
                return;
            }
            if (sizeMap.get(na) > sizeMap.get(nb)) {
                parents.put(nb, na);
                sizeMap.put(na, sizeMap.get(na) + sizeMap.get(nb));
                sizeMap.remove(nb); // 删除小的节点
            } else {
                parents.put(na, nb);
                sizeMap.put(nb, sizeMap.get(na) + sizeMap.get(nb));
                sizeMap.remove(na); // 删除小的节点
            }
        }
        Node findFather(V a) {
            Node n = nodes.get(a);
            while (parents.get(n) != n) {
                n = parents.get(n);
            }
            return n;
        }
    }
}
```

```java
    public static class UnionSet3{
        int[] parents; // p[1] = 3  1的爸爸是3
        int[] size; // i所在的集合大小是多少，i必须是代表节点
        int[] help; //栈
        int sets; // 多少个数据？
        public UnionSet3(int N) {
            this.parents = new int[N];
            this.size = new int[N];
            this.help = new int[N];
            this.sets = N;
            for (int i = 0; i < N; i++) {
                parents[i] = i;
                size[i] = 1;
            }
        }
        // 同时需要路径压缩
        private int find(int i ){
            int hi = 0;
            while(i != parents[i]){
                help[hi++] = i; // 栈
                i = parents[i];
            }
            for(hi--; hi >= 0; hi--){
                parents[help[hi]] = i; // 路径压缩，所有子节点都是指向i
            }
            return i;
        }
        void union(int a, int b){
            int i = find(a);
            int j = find(b);
            if(i != j){
                if(size[i] >= size[j]){
                    size[i] += size[j];
                    parents[j] = i;
                }else{
                    size[j] += size[i];
                    parents[i] = j;
                }
                sets--;
            }
        }
    }
```

200. 岛屿数量
给你一个由 '1'（陆地）和 '0'（水）组成的的二维网格，请你计算网格中岛屿的数量。
岛屿总是被水包围，并且每座岛屿只能由水平方向和/或竖直方向上相邻的陆地连接形成。
此外，你可以假设该网格的四条边均被水包围。
【感染函数】**复杂度：O（m*n）**
```java
    public int numIslands(char[][] grid) {
        int islands = 0;
        for (int i = 0; i < grid.length; i++) {
            for (int j = 0; j < grid[i].length; j++) {
                if(grid[i][j] == '1'){
                    islands++;
                    infect(grid, i, j);
                }
            }
        }
        return islands;
    }
    // 从(i,j)开始将所有连成一片的‘1’字符，变成数值0
    public static void infect(char[][] board, int i, int j){
        if(i < 0 || i == board.length || j < 0 || j == board[i].length || board[i][j] != '1'){
            return;
        }
        board[i][j] = 0;
        infect(board, i+1, j);
        infect(board, i-1, j);
        infect(board, i, j+1);
        infect(board, i, j-1);
    }
```
